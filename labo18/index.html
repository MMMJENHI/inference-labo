# Analyse Statistique Skalle-2

Cette cellule calcule les métriques de stationnarité et de gain du circuit en temps réel.

import numpy as np
import statio_lib as sk

# Récupération état circuit Skalle-2
params = sk.get_hardware_config()
signal = sk.get_buffer()

# Calcul Statistique Master Statio
rms = np.sqrt(np.mean(signal**2))
gain_db = 20 * np.log10(params.total_gain)

print(f"RMS: {rms:.4f}V | Gain: {gain_db:.2f}dB")
